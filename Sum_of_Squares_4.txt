function Sum_two_number(int $n){
	for($i=1; $i<=$n-1; $i++){
		$a=pow($i,2);
    	$F=$n-$a;
    	$F=sqrt($F);
		$s=is_decimal($F);
		if(($s==0)&&($F<$i)){
        	return array($i, $F);
    	}
	}
}
function is_decimal($val){
	return is_numeric($val) && floor($val) != $val;
